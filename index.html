Insertion Sort

function insertion_sort(arr, lb, ub) { for (let i = lb + 1; i <= ub; i++) { let current = arr[i]; let position = i - 1; while (position >= lb && arr[position] > current) { arr[position + 1] = arr[position]; position--; } arr[position + 1] = current; } }

Merge Sort

function merge_sort(arr) { if (arr.length < 2) return arr; let mid = Math.floor(arr.length / 2); let left = merge_sort(arr.slice(0, mid)); let right = merge_sort(arr.slice(mid)); return merge(left, right); } function merge(left, right) { let result = [], i = 0, j = 0; while (i < left.length && j < right.length) { if (left[i] < right[j]) { result.push(left[i++]); } else { result.push(right[j++]); } } return result.concat(left.slice(i)).concat(right.slice(j)); }

Hybrid Sort

function hybrid_sort(arr, threshold) { if (arr.length <= threshold) { return insertion_sort(arr, 0, arr.length - 1); } else { let mid = Math.floor(arr.length / 2); let left = hybrid_sort(arr.slice(0, mid), threshold); let right = hybrid_sort(arr.slice(mid), threshold); return merge(left, right); } }
