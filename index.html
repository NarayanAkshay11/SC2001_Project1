Hybrid Sorting Algorithm: Merge Sort + Insertion Sort

A one-page interactive demo showcasing the performance of Merge Sort, Insertion Sort, and the Hybrid Algorithm

Project Description

The hybrid Merge Sort and Insertion Sort algorithm combines the efficiency of Merge Sort for large datasets and Insertion Sort for smaller subarrays. The purpose of this project is to explore how using Insertion Sort for small subarrays can improve the overall performance.

Algorithms Explained

Merge Sort


function mergeSort(arr) {
    if (arr.length <= 1) return arr;
    const mid = Math.floor(arr.length / 2);
    const left = mergeSort(arr.slice(0, mid));
    const right = mergeSort(arr.slice(mid));
    return merge(left, right);
}
function merge(left, right) {
    let result = [], i = 0, j = 0;
    while (i < left.length && j < right.length) {
        if (left[i] < right[j]) result.push(left[i++]);
        else result.push(right[j++]);
    }
    return [...result, ...left.slice(i), ...right.slice(j)];
}

Insertion Sort


function insertionSort(arr) {
    for (let i = 1; i < arr.length; i++) {
        let key = arr[i], j = i - 1;
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
    }
    return arr;
}

Hybrid Merge + Insertion Sort


function hybridSort(arr, threshold) {
    if (arr.length <= threshold) return insertionSort(arr);
    const mid = Math.floor(arr.length / 2);
    const left = hybridSort(arr.slice(0, mid), threshold);
    const right = hybridSort(arr.slice(mid), threshold);
    return merge(left, right);
}
